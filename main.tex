\subsection{Introduction}
\begin{frame}{Introduction}
 Arrow defines two ways to make social choices in a capitalist democracy:
    \begin{itemize}    
    \item \textbf{Voting} (for political decisions)
    \item \textbf{The Market Mechanism} (for economic decisions)
    \end{itemize}
\end{frame}

\begin{frame}{Introduction}
 It also defines two ways to make choices based on a single entity, ideally respecting a single common will:
    \begin{itemize}
        \item \textbf{Dictatorship:} The social preference is determined entirely by the preferences of a single individual.
        \item \textbf{Convention:} Collective choices are dictated by a divine or commonly shared will.
    \end{itemize}
    \vspace{0.5 cm}
     In these societies, \textbf{there are no conflicting choices}: \\
     \vspace{0.25 cm}
     \begin{itemize}
        \item[$\Rightarrow$] \underline{Any single entity can be rational in his choice.}
    \end{itemize}
\end{frame}

\begin{frame}{The Core Problem}
    \begin{itemize}
        \item \textbf{How can we aggregate individual preferences into a collective, \uline{rational} social choice?}
        \begin{itemize}
            \item[$\Rightarrow$] We need to define a \textbf{\uline{social welfare function}}
        \end{itemize}
        \vspace{0.25 cm}
        \item Traditional democratic methods, often fail..
        \begin{itemize}
            \item Majority Decision
            \item Pareto optimum
            \item Compensation Principle
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Majority decision}
    \textbf{The Paradox of Voting:}\\
      It shows that even if every individual has rational, consistent preferences, the \textit{collective} social preference can be irrational.
    \vspace{0.5 cm}
    \begin{itemize}
        \item Three voters (1, 2, 3) ranking three options (A, B, C):
        \begin{enumerate}
            \item $A > B > C$
            \item $B > C > A$
            \item $C > A > B$
        \end{enumerate}
        \item \textbf{The Majority Result:} 
        \begin{itemize}
            \item A is preferred over B
            \item B is preferred over C
        \end{itemize}
        \item[$\Rightarrow$] Logically, A should be preferred over C; however:
        \begin{itemize}
            \item C is preferred over A $\rightarrow$ \textbf{Paradox}
        \end{itemize}
    \end{itemize}

    \note{
        \vfill
        \begin{center}
            disarmament, cold war, hot war
        \end{center}
        \vfill
    }
    
\end{frame}

\begin{frame}{Welfare economics}
    {\setlength{\leftskip}{0.9 cm}}
    If we view rationality as a maximization process, the challenge of achieving a \textit{social maximum} coincides with the central problem of welfare economics.
    \vspace{0.25 cm}
    \begin{itemize}
        \item \textbf{Goal:} Maximize social utility
        \vspace{0.25 cm}
        \item \textbf{Weaker goal:} Pareto optimum
        \vspace{0.25 cm}
        \begin{itemize}
            \item This solution still has problems:
            \begin{enumerate}
                \item Redistribution problem
                \item Preservation of Status Quo
                \item No social indifference map $\rightarrow$ No prescription for actions
            \end{enumerate}
        \end{itemize}
    \end{itemize}

        \end{center}

\end{frame}


\begin{frame}{Compensation principle}
    When choosing between 2 economic states \textit{x} and \textit{y}, \textit{x} should be preferred over \textit{y} if there is a method of paying compensations under \textit{x} so that everybody is made better off than they are in state \textit{y}.
    \vspace{0.25 cm}
    \begin{itemize}
        \item \textbf{Paradox:} It is possible that \textit{x} is preferred over \textit{y} and vice versa.
        \item Even without the paradox occurring \uline{it fails to produce a linear ordering of social choices.}
    \end{itemize}

\begin{frame}{A general problem}
    The aim of the paper is to show that the difficulties seen in voting and economics are general:
    \begin{itemize}
        \item For any methods of deriving social choices by aggregating individual preference patterns $\rightarrow$ it is possible to find individual preference patterns that create a social choice which does not have a linear ordering.
        \item It is assumed that each individual behaves rationally $\rightarrow$ his behavior can be described by an indifference map.
    \end{itemize}

\subsection{Definitions and Notation}
\begin{frame}{Basic definitions}
    \begin{itemize}
        \item We define a set of mutually exclusive alternatives denoted by small letters, \textit{x, y, z...}
        \item A \textit{subset S} over the \textit{Set} contains the voting alternatives.
        \item User can only express a preference on a single pair of alternatives.
        \item Given a pair \textit{x, y} the chooser can express 3 possible choices:
        \begin{enumerate}
            \item Prefers \textit{x} over \textit{y}
            \item Neither \textit{x} or \textit{y} is preferred over the other 
            \item Prefers \textit{y} over \textit{x}
        \end{enumerate}
        \vspace{0.25 cm}
        \item To simplify relations between alternatives a single relation is defined:
        \textit{xRy} $\rightarrow$ x is preferred or indifferent to y
        
    \end{itemize}
\end{frame}

\begin{frame}{The decision Protocol}
    \begin{itemize}
        \item The chooser evaluates all possible pairs of alternatives to establish a global preference scale, without yet knowing the opportunity set \textit{S}.
        \item Once the subset \textit{S} is revealed, the chooser picks the alternative that ranks highest according to their prior ordering.
        \vspace{0.25 cm}
        \item \underline{Decisions are assumed to be consistent with one another.}
        \begin{itemize}
            \item If x preferred over y and y over z then x must be preferred over z.
            \item If x indifferent to y and y to over z then x must be indifferent to z.
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Axiom I, II}

    \begin{itemize}
        \item \textbf{Axiom I:} For all \textit{x} and \textit{y}, either \textit{xRy} or \textit{yRx}.
        \begin{itemize}
            \item Note that: both \textit{xRy} and \textit{yRx} can hold at the same time $\rightarrow$ \textit{x} is indifferent to \textit{y}.
            \item It holds even if \textit{(x = y)} implying \textit{xRx}.
        \end{itemize}
        \vspace{0.5 cm}
        \item \textbf{Axiom II:} For all \textit{x}, \textit{y}, and \textit{z}, \textit{xRy} and \textit{yRz} imply \textit{xRz}.
        \begin{itemize}
            \item Guarantees transitivity, hence consistency.
            \item Along with Axiom I, it guaranties weak ordering.
        \end{itemize}
    \end{itemize}

    \note{
        \vfill
        \begin{center}
            - Axiom I states that is always possible to compare 2 alternatives
        \end{center}
        \vfill
    }
    
\end{frame}

\begin{frame}{Definition I, II}
    \begin{itemize}
        \item \textbf{Definition I:} The notation \textit{xPy} is defined to mean not \textit{yRx}. 
        \begin{itemize}
            \item It is read as "\textit{x} is preferred to \textit{y}".
        \end{itemize}
        \vspace{0.5 cm}
        \item \textbf{Definition II:} The notation \textit{xIy} it means \textit{xRy} and \textit{yRx} 
        \begin{itemize}
            \item It is read as "\textit{x} is indifferent to \textit{y}"
        \end{itemize}
        \vspace{0.5 cm}
         \item We can now define 6 very intuitive \textbf{Lemmas:}
            \begin{itemize}
                \item \textbf{a:} For all \textit{x}, \textit{xRx}.
                \item \textbf{b:} If \textit{xPy}, then \textit{xRy}.
                \item \textbf{c:} If \textit{xPy} and \textit{yPz}, then \textit{xPz}.
                \item \textbf{d:} If \textit{xIy} and \textit{yIz}, then \textit{xIz}.
                \item \textbf{e:} For all \textit{x} and \textit{y}, either \textit{xRy} or \textit{yPx}.
                \item \textbf{f:} If \textit{xPy} and \textit{yRz}, then \textit{xPz}.
            \end{itemize}
        \end{itemize}

\begin{frame}{The Mechanics of Rational Choice}
    \begin{itemize}
        \item \textbf{Clear Terminology:} To avoid mathematical ambiguity:
        \begin{itemize}
            \item \textit{R} is defined strictly as a \textbf{"weak ordering"}.
            \item \textit{P} is defined strictly as a \textbf{"preference relation"}.
        \end{itemize}
        \vspace{0.5 cm}
        \item \textbf{Reduction to Pairs:} A rational individual's choice from \textit{any} collection of alternatives can be entirely determined just by knowing their choices between simple pairs.
    \end{itemize}
\end{frame}

\begin{frame}{The Ordering of social states}

            \item The amount of resources invested in each activity
            \item Each diplomatic and collective activity
        \end{enumerate}
        \begin{itemize}
            \item[$\Rightarrow$] Basically a snapshot of the whole society.
        \end{itemize}
        \item Each citizen has a definite ordering of all conceivable social states which they can order by whatever standards they deem relevant.
    \end{itemize}

\begin{frame}{Tastes vs Values}
    Arrow overcomes pure individualism by distinguishing between \textbf{Tastes} and \textbf{Values} of each individual.
    \vspace{0.5 cm}
    \begin{itemize}
        \item The Market takes into account only the ordering according to tastes, which cannot result in a true social maximum.
        \item Each preference of an individual should be considered, even the ones that do not directly relate to him.
        \begin{itemize}
            \item[$\Rightarrow$] Deciding whether a specific preference is relevant to society or simply "none of one's business" is, in itself, a value judgment. Thus, society must consider the entire system of values.

\subsection{The social welfare function}

\begin{frame}{Social welfare function}
    \framesubtitle{Bergson's Formulation}
    \begin{itemize}
        \item[$\Rightarrow$] Bergson assign to each \textbf{social state} a \underline{numerical social utility}, then it chooses the one yielding the higher welfare within the environment.
        \vspace{0.75 cm}
        \item This maximization approach do not require to have any measurability, it just need a relative ordering between the alternatives.
        \vspace{0.25 cm}
        \item Each relative ranking between two states should be varying with the variation of values from at least some individual; but what happens if this is not the case?  
    \end{itemize}
\end{frame}

\begin{frame}{Two Philosophies of the "Social Good"}
    \begin{itemize}
        \item \textbf{The Platonic View} 
        \begin{itemize}
            \item[$\Rightarrow$] Assumes it exist an \textbf{objective social good}.
            \item The social ordering remains exactly the same for every set of individual orderings.
            \item Historically used to justify governance by an elite.
        \end{itemize}
        \vspace{0.3 cm}
        \item \textbf{The Utilitarian View} 
        \begin{itemize}
            \item Assert the social good is purely a \textbf{composition of individual goods}.
            \item This point of view is pushed even further with the hedonist psicology $\rightarrow$ individual good = individual desires.
            \item[$\Rightarrow$] Hence the global social ordering relation ($R$) is entirely determined by the individual orderings ($R_1, \dots, R_n$).
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Social welfare function}
    \framesubtitle{Arrow's definition}
    \textbf{Formal Definition} A "social welfare function" is a process which, for each set of individual orderings ($R_1, \dots, R_n$) for alternative social states, yields a corresponding social ordering of alternative social states ($R$).
    \vspace{0.25 cm}
    \begin{itemize}
        \item Arrow's individual orderings refer to \textbf{values}, whereas Bergson relies on \textbf{tastes} $\rightarrow$ assumes \uline{individual values already produce a social value judgment}.
        \vspace{0.25 cm}
        \item Arrow's function is essentially a method for choosing \textit{which} specific Bergson-type social welfare function should be applied to the society item $\blacktriangleright$ Meta-Rule
        %\item Bergson's individual tastes are represented by utility functions (which are essentially ordering relations), both functions act as a rule for assigning a global social ordering based on individual orderings.%
    \end{itemize}
\end{frame}

\begin{frame}{Minimal Scope of the Proof}
    \begin{itemize}
        \item \textbf{2 individuals} $\rightarrow$ \textit{\{1, 2\}}
        \begin{itemize}
            \item Better than 3
            \item Way simpler proof
        \end{itemize}
        \vspace{0.25 cm}
        \item \textbf{3 alternatives} $\rightarrow$ \textit{\{x, y, z\}}
        \vspace{0.25 cm}
        \item Results will still hold for any numer of individual
        \begin{itemize}
            \item[$\Rightarrow$]  Since the final theorem proves an \textit{impossibility}, if the system fails in this minimal scenario, it will inevitably fail for larger populations and more alternatives.
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{The Five Conditions}
    \framesubtitle{Condition 1: Admissible Orderings}
    \textbf{Condition 1:} The social welfare function must be defined for every admissible pair of individual orderings ($R_1, R_2$) and must give rise to a true, rational social ordering.
    \begin{itemize}
        \item[$\Rightarrow$] The Domain of the Function
    \end{itemize}
    \begin{itemize}
        \vspace{0.5 cm}
        \item Arrow does not demand that the function processes every possible set of individual orderings.
        \item He focuses on "admissible" orderings, which are based on realistic, empirical human behavior.
        
    \end{itemize}
\end{frame}

\begin{frame}{The Five Conditions}
    \framesubtitle{Condition 2: Positive Association of Values}
    \textbf{Condition 2:} If an alternative social state $x$ rises or does not fall in the ordering of each individual without any other changes, and if $x$ was already preferred to $y$, then $x$ must remain preferred to $y$.
    \begin{itemize}
        \item[$\Rightarrow$] Reflecting "welfare" not "illfare"
    \end{itemize}
    \vspace{0.5 cm}
    \begin{itemize}
        \item The social welfare function must respond accordingly to alterations in individual values.
        \item Voting for an alternative should have the usual effect of making it more likely to be adopted.
    \end{itemize}
\end{frame}

\begin{frame}{The Five Conditions}
    \framesubtitle{Condition 3: Independence of Irrelevant Alternatives}
    \textbf{Condition 3:} Let $R_1, R_2$ and $R'_1, R'_2$ be two sets of individual orderings. If, for both individuals $i$ and for all $x$ and $y$ in a given set of alternatives $S$, $xR_iy$ if and only if $xR'_iy$, then the social choice made from $S$ is the same whether the individual orderings are $R_1, R_2$ or $R'_1, R'_2$.
    \begin{itemize}
        \item[$\Rightarrow$] Avoiding Interferences
    \end{itemize}
    \vspace{0.5 cm}
    \begin{itemize}
        \item The choice made by society from any given set of alternatives must depend \textit{only} on the individuals' orderings among those specific alternatives $\rightarrow$ Methods of social choice are of the type of voting.
        \item The existence or removal of an "irrelevant" alternative outside the given set must not alter the social choice.
        \item \textit{Example:} Clubs voting system.
    \end{itemize}
\end{frame}

\begin{frame}{The Five Conditions}
    \framesubtitle{Condition 4: Citizens' Sovereignty}
    \textbf{Condition 4:} The social welfare function is not to be imposed.
    \begin{itemize}
        \item[$\Rightarrow$] Freedom of Choice
    \end{itemize}
    \vspace{0.5 cm}
    \begin{itemize}
        \item \textbf{Definition 4:} A social welfare function will be said to be imposed if for some pair of distinct alternatives $x$ and $y$, $xRy$ for any set of individual orderings $R_1, R_2$.
        \item A function is imposed if society can never express a preference for a certain alternative over another, regardless of the individuals' tastes. 
    \end{itemize}
\end{frame}

\begin{frame}{The Five Conditions}
    \framesubtitle{Condition 5: Nondictatorship}
    \textbf{Condition 5:} The social welfare function is not to be dictatorial.
    \begin{itemize}
        \item[$\Rightarrow$] Excluding Autocracy
    \end{itemize}
    \vspace{0.5 cm}
    \begin{itemize}
        \item \textbf{Definition 5:} A social welfare function is said to be dictatorial if there exists an individual $i$ such that for all $x$ and $y$, $xP_iy$ implies $xPy$ regardless of the orderings of all individuals other than $i$.
   
        \item A function is dictatorial if the strict preference of one specific individual $x P_i y$ forces society to strictly prefer $x$ to $y$ $\Rightarrow$ $xPy$ .
    \end{itemize}
\end{frame}

\subsection{The possibility theorem}

\begin{frame}
    \vfill
    \begin{center}
        \usebeamercolor[fg]{title} 
        \LARGE \textbf{THE POSSIBILITY THEOREM} \\ 
        \vspace{0.25 cm} 
        \Large \textit{FOR \\ SOCIAL WELFARE \\ FUNCTIONS}
    \end{center}
    \vfill
\end{frame}

\begin{frame}{Individual Orderings}
    Arrow adopts two standard "individualistic assumptions" from welfare economics:
    \begin{enumerate}
        \item \textbf{Self-Interest:} An individual's choice depends only on what they get.
        \item \textbf{More is Better:} An individual will always prefer a state where they receive more of at least one commodity.
    \end{enumerate}
    \vspace{0.25 cm}
    \textbf{The Consequence:}
    \begin{columns}[T, onlytextwidth]
        \begin{column}{0.5\textwidth}
            \begin{figure}
                \centering
                \includegraphics[width=\linewidth]{table.png}
            \end{figure}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{itemize}
                \vspace{0.25 cm}
                \item No "best" option $\Rightarrow$ all orderings are \textbf{"admissible"}.
                \begin{itemize}
                    \item[$\Rightarrow$] Social welfare function must be able to process \textit{every} possible pair of individual orderings ($R_1, R_2$).
                \end{itemize} 
            \end{itemize}
        \end{column}
        
    \end{columns}
\end{frame}

\begin{frame}{Setting Up the Proof}
    \textbf{The Objective:}
    \vspace{0.25 cm}
    \begin{quote}
        ``Some consequences will be drawn from Conditions 1-5 for the present case of a social welfare function for two individuals and three alternatives. It will be shown that the supposition that there is a social welfare function satisfying those conditions leads to a contradiction.''
    \end{quote}
    \vspace{0.5 cm}
    \textbf{Notation:}
    \begin{itemize}
        \item Let $x, y$, and $z$ be the three alternatives among which choice is to be made, (three possible distributions of commodities).
        \item Let $x'$ and $y'$ be variable symbols which represent possible alternatives, (range over the values $x, y, z$).
    \end{itemize}
\end{frame}

\begin{frame}{Deriving the Consequences}
    \framesubtitle{Consequence 1: Unanimity}
    \begin{itemize}
        \item \textbf{Definition:} If $x'P_1y'$ and $x'P_2y'$, then $x'Py'$.
        \item[$\Rightarrow$] If both individuals strictly prefer $x'$ to $y'$, then society must strictly prefer $x'$ to $y'$.
        \vspace{0.25 cm}
        \item \textbf{Proof:} By Condition 4, society is allowed to prefer $x'$ to $y'$ in at least one hypothetical scenario ($R_1'$, $R_2'$). If we take that scenario and raise $x'$ to the very top of both individuals' rankings ($R_1''$, $R_2''$), Condition 2 dictates that society must still prefer $x'$ to $y'$. Because of Condition 3, the choice between $x'$ and $y'$ depends only on their relative rankings, so this social preference holds \textit{whenever} both individuals agree on $x'$ over $y'$.

    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{The Possibility Theorem}
    \framesubtitle{Consequence 2: The Prevailing Will}

    \begin{itemize}
        \item \textbf{Definition:} Suppose that for some $x'$ and $y'$, whenever $x'P_1y'$ and $y'P_2x'$, society prefers $x'Py'$. Then, for that $x'$ and $y'$, whenever $x'P_1y'$, $x'Py'$.
        \item[$\Rightarrow$] If in a given choice, the will of Individual 1 prevails against the direct opposition of Individual 2, then Individual 1's views will certainly prevail if Individual 2 becomes indifferent or agrees.
        \vspace{0.25 cm}
        \item \textbf{Proof:} This logically follows from Condition 2. If society already chooses $x'$ when Individual 2 completely opposes it, then if Individual 2 raises $x'$ in their ranking, $x'$ cannot mathematically lose ground socially.
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{The Possibility Theorem}
    \framesubtitle{Consequence 3: Opposing Interests}

    \begin{itemize}
        \item \textbf{Definition:} If $x'P_1y'$ and $y'P_2x'$, then $x'Iy'$.
        \item[$\Rightarrow$] If the two individuals have exactly opposing strict preferences between two given alternatives, then society must be indifferent between those alternatives.
        \vspace{0.25 cm}
        \item \textbf{Proof} \textit{(by contradiction)}\textbf{:} If society is \textit{not} indifferent, and instead 1's preference overrides 2's opposition. According to Consequence 2, individual 1 would then always win on that specific choice. \\
        \vspace{0.25 cm}
        Giving  1 the power to override the other on even a single pair of alternatives triggers a chain reaction that eventually makes individual 1 a dictator over \textit{all} choices. Since Condition 5 strictly forbids a dictator, the initial assumption must be false.
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{The Possibility Theorem}
    \framesubtitle{The Contradiction}
    
    \begin{itemize}
        \item \textbf{Individual 1 ordering:} $x, y, z$
        \item \textbf{Individual 2 ordering:} $z, x, y$
    \end{itemize}
    
    \vspace{0.25 cm}
    \textbf{The Contradiction:}
    \begin{enumerate}
        \item By \textbf{Consequence 1} ($xPy$).
        \item By \textbf{Consequence 3} ($yIz$) and ($xIz$).
        \item By \textbf{Axioms I, II} ($xPz$).
    \end{enumerate}
    
    \vspace{0.25cm}
    \begin{center}
        \textcolor{darkred}{\textbf{Contradiction:}} Society cannot simultaneously prefer $x$ to $z$ \textit{and} be indifferent between them.
    \end{center}
\end{frame}

\begin{frame}
    \frametitle{The Possibility Theorem}
    \begin{quote}$\blacktriangleright$
        ``\textbf{Possibility Theorem.} --- If there are at least three alternatives among which the members of the society are free to order in any way, then every social welfare function satisfying Conditions 2 and 3 and yielding a social ordering satisfying Axioms I and II must be either imposed or dictatorial.''
    \end{quote}
    \vspace{0.5 cm}
    \begin{itemize}
        \item \textbf{Conclusion:}
        \begin{quote}
        ``If no prior assumptions are made about the nature of individual orderings, there is no method of voting which will remove the paradox of voting.''
        \end{quote}
        \vspace{-0.25cm}
        \begin{itemize}
            \item[$\Rightarrow$] \textbf{Political Systems:} No voting system can be perfectly fair and perfectly rational at the same time.
            \item[$\Rightarrow$] \textbf{Economic Systems:} The free market cannot mathematically guarantee an optimal and rational social choice.
        \end{itemize}  
    \end{itemize}
\end{frame}

\subsection{Implications for Social Welfare Judgments}

\begin{frame}{Interpretation of the Theorem}

    \begin{itemize}
        \item Conditions 1 and 3 together exclude any \uline{interpersonal comparison of social utility}.
        \begin{itemize}
            \item[$\Rightarrow$] System inevitably break down, since it can only process rankings and cannot weigh the intensity of human feelings.
        \end{itemize}
        \vspace{0.5 cm}
        \item \textbf{Arrow then restate the conclusion as:}
        \begin{quote}
            ``If we exclude the possibility of interpersonal comparisons of utility, then the only methods of passing from individual tastes to social preferences which will be satisfactory [...] are either imposed or dictatorial''.
        \end{quote}
        \begin{itemize}
            \item[$\Rightarrow$] ``The doctrine of voters' sovereignty is incompatible with that of collective rationality''
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Interpretation of the Theorem}
    \textbf{How to move on?}
    \vspace{0.25 cm}
    \begin{itemize}
        \item \textbf{Relaxing the Conditions:} If we want a system that is neither imposed nor dictatorial, Arrow concludes that we must relax some of our initial rules.
        \vspace{0.5 cm}
        \item \textbf{Restricting Individual Values:} Since we still cannot magically measure interpersonal utility, the only way out is to abandon the requirement that the system works for a \textit{wide range of individual orderings}.
    \end{itemize}
\end{frame}

\begin{frame}{Reflections on the New Welfare Economics}

    \textbf{New Welfare Economics} focuses on finding maximal states, (Pareto optimum), based purely on individualistic assumptions.
    \begin{itemize}
        \item[$\Rightarrow$] Mathematically proven faulty by Arrow
    \end{itemize}
    \vspace{0.5 cm}
    \textbf{Need of Welfare Economics:} 
    \begin{itemize}
        \item However, if we add further ethical restrictions to the individualistic ones, the true social optimum will still be found among those individualistic maximal states.
        \begin{itemize}
            \item[$\Rightarrow$] Traditional welfare economics still can be used to restrict the range in which society must look for the ultimate social maximum.
        \end{itemize}
    \end{itemize}     
\end{frame}

\begin{frame}{The Necessity of shared Ethical Norms}
    The mathematical failure of pure individualism proves that \uline{there must be a divergence between private desires and collective social benefits}.\\
    \vspace{0.5 cm}
    \textbf{The Ethical addition:} $\rightarrow$ Values
    \begin{itemize}
        \item To make a rational social choice possible, society must incorporate a scheme of \textbf{socio-ethical norms}, which \uline{must be sufficiently similar among members of society} $\rightarrow$ (Impossibility otherwise). 
        \vspace{0.25 cm}
        \item[$\Rightarrow$] A democratic machine \uline{capable of collective rationality} can only function without collapsing into contradiction or dictatorship if its citizens hold socio-ethical norms that are \textbf{``sufficiently similar''}.  
    \end{itemize}
\end{frame}

\begin{frame}{A One-Commodity World}

\textbf{What if, in the world, there exists only one commodity?}

    \begin{itemize}
        \item If we keep considering (1), (2) of Part IV any individual have only one possible ordering of the social states.
        \begin{itemize}
            \item[$\Rightarrow$] Conditions 2, 3, and 4 irrelevant, Condition 5 is weaker.
        \end{itemize}
        \item \textbf{Arbitrary Weak Ordering:} Any specification of a social ordering which does not coincide completely with the ordering of any one individual will be a social welfare function compatible with all the conditions.
        \item \textbf{Difficulty in a multi-commodity world:} We cannot compare the total production in two different social states without a standard of value (e.g. prices). Since this standard is dependent on the distribution of income, the total output inherently relies on ethical judgments.
    \end{itemize}
\end{frame}

\begin{frame}{Distributional Ethics combined with Individualism}
    \textbf{A Proposed Ethical Schema:} Some economists attempt to solve economic problems by assuming two \uline{value judgments}:
    \begin{enumerate}
        \item If everybody is better off according to their own \textit{tastes}, society prefers that state.
        \item There is a universally accepted ordering of different welfare distributions (usually egalitarian).
    \end{enumerate}
    \vspace{0.25 cm}
    \begin{itemize}
        \item This approach requires a sharp distinction between Values and Tastes: 
        \begin{itemize}
            \vspace{-0.25 cm}
            \item[$\Rightarrow$] All individuals are assumed to share the same \textit{values} regarding equitable distribution, while having, totally opposite, purely individualistic \textit{tastes}.
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Failure of Individualistic Ethics}
    \framesubtitle{The Bergson SWF}

    \begin{itemize}
        \item \textbf{The Bergson social welfare function:} \\
        Assigns a social ordering to each possible set of individual orderings representing purely personal tastes, which since are individualistic is mathematically isomorphic to the Possibility Theorem, \textit{since it was based on individualistic assumptions (1), (2) of Part IV.}
        \vspace{0.5 cm}
        \item \textbf{The Conclusion:}
        \begin{quote}
            ``Essentially, the two value judgments amount to erecting individualistic behavior into a value judgment. It is not surprising, then, that such ethics can be no more successful than the actual practice of individualism.''
        \end{quote}
    \end{itemize}
